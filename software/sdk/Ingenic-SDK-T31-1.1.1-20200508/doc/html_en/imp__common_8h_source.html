168 static inline int calc_pic_size(int width, int height, IMPPixelFormat imp_pixfmt)
169 {
170  int bpp1 = 0, bpp2 = 1,size;
171 
172 #define BPP(FMT, A, B) case FMT: bpp1 = A;bpp2 = B;break
173  switch (imp_pixfmt) {
174  BPP(PIX_FMT_NV12, 3, 2);
175  BPP(PIX_FMT_YUYV422, 2, 1);
176  BPP(PIX_FMT_UYVY422, 2, 1);
177  BPP(PIX_FMT_RGB565BE, 2, 1);
178  BPP(PIX_FMT_BGR0, 4, 1);
179  default: break;
180  }
181 #undef BPP
182  size = width * height * bpp1 / bpp2;
183 
184  return size;
185 }
